class Stack {
  constructor(...args) {
    // 使用数组进行模拟
    this.stack = [...args];
  }
  push(...items) {
    // 入栈
    return this.stack.push(...items);
  }
  pop() {
    // 出栈
    return this.stack.pop();
  }
  peek() {
    // 栈顶
    return this.isEmpty() ? undefined : this.stack[this.size() - 1];
  }
  isEmpty() {
    // 栈是否为空
    return this.size() == 0;
  }
  size() {
    // 栈长度
    return this.stack.length;
  }
  clear() {
    // 清空栈
    this.stack.length = 0;
  }
}

链表：Linked Lists

  • 单链表：单链表是维护一系列节点的数据结构，其特点是：每个节点包含了数据，同时包含指向链表中下一个节点的指针。
  • 双向链表：不同于单链表，双向链表特点：每个节点分支除了包含其数据以外，还包含了分别指向其前驱和后继节点的指针。

实现

class Node {
  constructor(data) {
    // 为当前节点储存的数据
    this.data = data;
    // next指向下一个节点
    this.next = null;
    // prev指向前一个节点
    this.prev = null;
  }
}

class DoublyLinkedList {
  constructor() {
    // 双向链表开头
    this.head = null;
    // 双向链表结尾
    this.tail = null;
  }
  add(item) {
    // 实例化一个节点
    let node = new Node(item);
    if (!this.head) {// 如果当前链表还没有头
      this.head = node;
      this.tail = node;
    } else {// 如果当前链表已经有了头
      // 把当前的尾部作为新节点的prev
      node.prev = this.tail;
      // 把当前的尾部的next指向为新节点node
      this.tail.next = node;
      // 把当前的新节点作为尾部
      this.tail = node;
    }
  }
  addAt(index, item) {
    // 链表长度
    let length = this.length();
    // index校验
    if (index < 0) return;
    // 链表为空且index为有效值
    if (this.isEmpty() && index === 0) {
      this.add(item);
      return;
    }
    // 维护查找时当前节点的指针
    let current = this.head;
    // 维护查找时当前节点的索引
    let counter = 1;
    let node = new Node(item);
    // 在头部插入(边界)
    if (index === 0) {
      this.head.prev = node;
      node.next = this.head;
      this.head = node;
      return;
    }
    while (current) {
      // 在尾部插入(边界)
      if (index === length) {
        this.add(item);
        return;
      }
      current = current.next;
      // 既不在头部也不在尾部
      if (counter === index) {
        node.prev = current.prev;
        current.prev.next = node;
        node.next = current;
        current.prev = node;
      }
      counter++;
    }
  }
  remove(item) {
    let current = this.head;
    while (current) {
       // 找到了目标节点
      if (current.data === item) {
        if (current == this.head && current == this.tail) {// 删除既是头部又是尾部(边界)
          this.head = null;
          this.tail = null;
        } else if (current == this.head) {// 删除为头部(边界)
          this.head = this.head.next;
          this.head.prev = null;
        } else if (current == this.tail) {// 删除为尾部(边界)
          this.tail = this.tail.prev;
          this.tail.next = null;
        } else {// 删除既不是头部也不是尾部
          current.prev.next = current.next;
          current.next.prev = current.prev;
        }
      }
      current = current.next;
    }
  }
  removeAt(index) {
    let length = this.length();
    // index校验
    if (index < 0) return;
    // 链表为空
    if (this.isEmpty()) return;
    let current = this.head;
    let counter = 1;
    if (index === 0) {
      if (length === 1) {// 删除既是头部又是尾部(边界)
        this.head = null;
        this.tail = null;
        return;
      } else {// 删除为头部(边界)
        this.head = this.head.next;
        this.head.prev = null;
      }
    } else {
      while (current) {
        if (index === length - 1) {// 删除为尾部(边界)
          this.tail = this.tail.prev;
          this.tail.next = null;
          return;
        } else if (index > length - 1) {// index为有效值
          return;
        }
        current = current.next;
        // 删除既不是头部也不是尾部
        if (counter === index) {
          current.prev.next = current.next;
          current.next.prev = current.prev;
          break;
        }
        counter++;
      }
    }
  }
  reverse() {
    let current = this.head;
    let prev = null;
    while (current) {
      let next = current.next;
      // 前后倒置
      current.next = prev;
      current.prev = next;
      prev = current;
      current = next;
    }
    this.tail = this.head;
    this.head = prev;
  }
  swap(index1, index2) {
    // index校验
    if (index1 < 0 || index2 < 0) {
      return;
    }
    // 使index1始终小于index2，方便后面查找交换
    if (index1 > index2) {
      return this.swap(index2, index1);
    }
    let current = this.head;
    let counter = 0;
    let firstNode;
    while (current !== null) {
      if (counter === index1) {// 保存找到的第一个节点
        firstNode = current;
      } else if (counter === index2) {// 找到的第二个节点，进行交换
        let temp = current.data;
        current.data = firstNode.data;
        firstNode.data = temp;
      }
      current = current.next;
      counter++;
    }
    return true;
  }
  isEmpty() {
    // 链表是否为空
    return this.length() < 1;
  }
  length() {
    let current = this.head;
    // 链表长度
    let counter = 0;
    // 完整遍历链表
    while (current !== null) {
      counter++;
      current = current.next;
    }
    return counter;
  }
  traverse(fn) {
    let current = this.head;
    // 完整遍历链表，执行回调
    while (current !== null) {
      fn(current);
      current = current.next;
    }
    return true;
  }
  find(item) {
    let current = this.head;
    let counter = 0;
    // 完整遍历链表
    while (current) {
      if (current.data == item) {
        return counter;
      }
      current = current.next;
      counter++;
    }
    return false;
  }
}

树：Trees

  • 除了根节点以外，所有的节点都有一个父节点；
  • 每一个节点都可以有若干子节点，如果没有子节点，则称此节点为叶子节点；
  • 一个节点所拥有的叶子节点的个数，称之为该节点的度，因此叶子节点的度为 0；
  • 所有节点中，最大的度为整棵树的度；
  • 树的最大层次称为树的深度。

二叉搜索树

  • 左子树上所有结点的值均小于或等于它的根结点的值；
  • 右子树上所有结点的值均大于或等于它的根结点的值；
  • 左、右子树也分别为二叉搜索树。
class Node {
  constructor(data) {
    // left指向左节点
    this.left = null;
    // right指向右节点
    this.right = null;
    // 为当前节点储存的数据
    this.value = data;
  }
}

class BinarySearchTree {
  constructor() {
    this.root = null;
  }
  insert(value) {
    let newNode = new Node(value);
    // 判读是否存在根节点
    if (!this.root) {
      this.root = newNode;
    } else {
      this.insertNode(this.root, newNode);
    }
  }
  insertNode(root, newNode) {
    // 根据待插入节点的值的大小，递归
    if (newNode.value < root.value) {
      !root.left ? (root.left = newNode) : this.insertNode(root.left, newNode);
    } else {
      !root.right
        ? (root.right = newNode)
        : this.insertNode(root.right, newNode);
    }
  }
  remove(value) {
    if (this.root) {
      this.removeNode(this.root, value);
    }
  }
  removeNode(root, value) {
    if (!root) {
      return null;
    }
    if (value < root.value) {
      root.left = this.removeNode(root.left, value);
      return root;
    } else if (value > root.value) {
      root.right = this.removeNode(root.right, value);
      return root;
    } else {
      // 当前root节点无左右子节点(边界)
      if (!root.left && !root.right) {
        root = null;
        return root;
      }
      if (root.left && !root.right) {// 当前root节点只有左子节点(边界)
        root = root.left;
        return root;
      } else if (root.right) {// 当前root节点只有右子节点(边界)
        root = root.right;
        return root;
      }
      // 找到右侧最小的节点
      let minRight = this.findMinNode(root.right);
      root.value = minRight.value;
      root.right = this.removeNode(root.right, minRight.value);
      return root;
    }
  }
  findMinNode(root) {
    if (!root.left) {
      return root;
    } else {
      return this.findMinNode(root.left);
    }
  }
  search(value) {
    if (!this.root) {
      return null;
    }
    return this.searchNode(this.root, value);
  }
  searchNode(root, value) {
    if (!root) {
      return null;
    }
    if (value < root.value) {
      return this.searchNode(root.left, value);
    } else if (value > root.value) {
      return this.searchNode(root.right, value);
    }
    return root;
  }
  preOrder(root) {
    // 前序遍历
    if (root) {
      console.log(root.value);
      this.preOrder(root.left);
      this.preOrder(root.right);
    }
  }
  inOrder(root) {
    // 中序遍历
    if (root) {
      this.inOrder(root.left);
      console.log(root.value);
      this.inOrder(root.right);
    }
  }
  postOrder(root) {
    // 后序遍历
    if (root) {
      this.postOrder(root.left);
      this.postOrder(root.right);
      console.log(root.value);
    }
  }
}

红黑树

  • 结点是红色或黑色。
  • 根结点是黑色。
  • 每个叶子结点都是黑色的空结点（NIL结点）。
  • 每个红色结点的两个子结点都是黑色。(从每个叶子到根的所有路径上不能有两个连续的红色结点)
  • 从任一结点到其每个叶子的所有路径都包含相同数目的黑色结点。

节点的插入

  • 插入的数据都是红色。
  1. 插入的位置是根节点：直接插入根节点位置，并且把颜色变为黑色
  2. 插入的位置父节点为黑色：直接插入即可
  3. 父节点、叔节点为红色，祖父节点为黑色：把叔，父变为黑，祖父变为红。（如果祖父节点的父节点为红色，则把祖父节点的树，看作一个新插进去的红色节点，进行递归处理）
  4. 父节点为红色，叔节点、祖父节点为黑色，插入的节点为父节点的左子节点：父变为黑，祖父变为红，以祖节点为旋转轴进行右旋转
  5. 父节点为红色，叔节点、祖父节点 为黑色，插入的节点为父节点的右子节点：以父节点为根进行左旋转，在安照情况4进行处理

图遍历的思路

  1. 每一个顶点有三种状态

    • 未发现（没有发现这个顶点）
    • 已经发现（发现其他的顶点连接到这里，但是没有去查找这个顶点的全部连接的顶点）
    • 已经探索（已经发现这个顶点连接的全部顶点）

  2. 记录顶点是否被访问过，使用三种颜色来反映他们的状态

    • 白色（未发现）
    • 灰色（已经发现）
    • 黑色（已经探索）

  3. 广度优先的遍历过程

    • 发现未发现顶点后，存放队列中，等待查找，并且将这些顶点标记未已发现
    • 在队列中拿出已经发现的顶点，开始探索全部顶点，并且要跳过已经探索的顶点
    • 遍历完这个顶点以后，将这个顶点标志为已经探索
    • 循环在队列中探索下一个顶点

  4. 深度优先的过程

    广度优先算法我们使用的是队列，深度优先的原理：使用递归

    • 从某一顶点开始查找，并且将这个定边标记为已经发现（灰色）
    • 从这个顶点开始探索其他的全部的顶点，并且跳过已经发现的顶点
    • 遍历完这个顶点以后，将这个顶点标记为已探索（黑色）
    • 递归返回，继续探索下一个路径的最深顶点
class Graph {
  constructor() {
    // 使用Map数据结构描述顶点关系
    this.AdjList = new Map();
  }
  addVertex(vertex) {
    if (!this.AdjList.has(vertex)) {
      // 述顶点与边的关系
      this.AdjList.set(vertex, []);
    } else {
      throw `'${vertex}' 已存在`;
    }
  }
  addEdge(vertex, node) {
    if (this.AdjList.has(vertex)) {
      if (this.AdjList.has(node)) {
        let arr = this.AdjList.get(vertex);
        if (!arr.includes(node)) {
          arr.push(node);
        }
      } else {
        throw `不能添加不存在的 '${node}'`;
      }
    } else {
      throw `需先添加 '${vertex}'`;
    }
  }
  print() {
    for (let [key, value] of this.AdjList) {
      console.log(key, value);
    }
  }
  createVisitedObject() {
    let map = {};
    for (let key of this.AdjList.keys()) {
      map[key] = false;
    }
    return map;
  }
  // 广度优先算法
  bfs(initialNode) {
    let visited = this.createVisitedObject();
    let queue = [];
    visited[initialNode] = true;
    queue.push(initialNode);
    while (queue.length) {
      let current = queue.shift();
      console.log(current);
      let arr = this.AdjList.get(current);
      for (let elem of arr) {
        if (!visited[elem]) {
          visited[elem] = true;
          queue.push(elem);
        }
      }
    }
  }
  // 深度优先算法
  dfs(initialNode) {
    let visited = this.createVisitedObject();
    this.dfsHelper(initialNode, visited);
  }
  dfsHelper(node, visited) {
    visited[node] = true;
    console.log(node);
    let arr = this.AdjList.get(node);
    for (let elem of arr) {
      if (!visited[elem]) {
        this.dfsHelper(elem, visited);
      }
    }
  }
}

class PrefixTree {
  constructor() {
    this.root = new Node();
  }
  addWord(str) {
    const addWordHelper = (node, str) => {
      if (str.length === 0) {
        return false;
      }
      // 将当前字符节点添加至子节点集合
      if (!node.children[str[0]]) {
        node.children[str[0]] = new Node(str[0]);
      }
      // 将子节点标记为叶子节点
      if (str.length === 1) {
        node.children[str[0]].isEnd = true;
      }
      addWordHelper(node.children[str[0]], str.slice(1));
    };
    addWordHelper(this.root, str);
  }
  predictWord(str) {
    // 收集相同前缀
    let allWords = [];
    // 获取相同前缀节点
    let getPrefixNode = function (node, str) {
      if (str.length === 0) {
        return node;
      } else {
        // 当前字符节点
        const child = node.children[str[0]];
        if (child) {
          return getPrefixNode(child, str.slice(1));
        } else {
          return null;
        }
      }
    };
    let allWordsHelper = function (node, string = "") {
      let children = node.children;
      if (node.isEnd && string.length !== 0) {
        allWords.push(str + string);
      }
      for (let k in children) {
        let word = string + children[k].value;
        allWordsHelper(children[k], word);
      }
    };
    let prefixNode = getPrefixNode(this.root, str);
    // 存在相同前缀
    if (prefixNode) {
      allWordsHelper(prefixNode);
    }
    return allWords;
  }
}

应用

  • 单词前缀

散列表（哈希表）：Hash Tables

  • hash就是把任意长度的输入（又叫做预映射，pre-image），通过散列算法，变换成固定长度的输出，该输出就是散列值。
  • 这种转换是一种压缩映射，也就是，散列值的空间通常远小于输入的空间，不同的输入可能会散列成相同的输出，所以不可能从散列值来唯一的确定输入值。简单的说就是一种将任意长度的消息压缩到某一固定长度的消息摘要的函数。
  • HASH主要用于信息安全领域中加密算法，它把一些不同长度的信息转化成杂乱的128位的编码,这些编码值叫做HASH值.也可以说，通俗的说hash就是找到一种数据内容和数据存放地址之间的映射关系
  • 哈希算法有一个很大的特点，就是你不能从结果推算出输入,所以又称为不可逆的算法

实现

class HashTable {
  constructor() {
    // 哈希表
    this.table = [];
  }
 
  // 哈希函数
  loseloseHashCode(key) {
    let hash = 0;
    for (let i = 0; i < key.length; i++) {
      hash += key[i].charCodeAt(); // 计算key unicode码
    }
    // 取模
    return hash % 37; // 减少碰撞
  }
   
  // 新增元素
  put(key, value) {
    // 获取key
    const position = this.loseloseHashCode(key);
    this.table[position] = value;
  }
 
  //  移除元素
  remove(key) {
    this.table[this.loseloseHashCode(key)] = undefined;
  }
 
  // 获取元素
  get(key) {
    return this.table[this.loseloseHashCode(key)];
  }
}
  • 开放定址法：一旦发生了冲突，就去寻找下一个空的散列地址，只要散列表足够大，空的散列地址总能找到
  • 再哈希法：又叫双哈希法，有多个不同的Hash函数，当发生冲突时，使用第二个，第三个，….，直到无冲突。虽然不易发生聚集，但是增加了计算时间
  • 链地址法：每个哈希表节点都有一个next指针，多个哈希表节点可以用next指针构成一个单向链表，被分配到同一个索引上的多个节点可以用这个单向链表连接起来
  • 建立公共溢出区：将哈希表分为基本表和溢出表两部分，凡是和基本表发生冲突的元素，一律填入溢出表
